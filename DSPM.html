def separate_files(folder_path, csv_path, real_folder, fake_folder):
    # Create the output folder 
    if not os.path.exists(real_folder):
        os.makedirs(real_folder)
   
    
    ids = set()
    with open(csv_path, 'r') as csv_file:
        reader = csv.reader(csv_file)
        for row in reader:
            ids.add(row[1]) 
   
   
    for filename in os.listdir(folder_path):
        file_path = os.path.join(folder_path, filename)
       
        # Check if the file is in the CSV file (matching the ID)
        file_id = os.path.splitext(filename)[0]
        if file_id in ids:
            # Move the file to the corresponding output folder
            output_path1 = os.path.join(real_folder, file_id + '.jpg')
            shutil.copy(file_path, output_path1)
            print(f"Moved {filename} to {output_path1}")
        else:
            output_path2 = os.path.join(fake_folder, file_id + '.jpg')
            shutil.copy(file_path, output_path2)
            print(f"Moved {filename} to {output_path2}")

import os
import shutil

def separate_files(folder_path, data, real_folder, fake_folder):
    # Create the output folders
    if not os.path.exists(real_folder):
        os.makedirs(real_folder)
    if not os.path.exists(fake_folder):
        os.makedirs(fake_folder)
    
    ids = set(data['ID']) 
   
    for filename in os.listdir(folder_path):
        file_path = os.path.join(folder_path, filename)
       
        # Check if the file is in the DataFrame (matching the ID)
        file_id = os.path.splitext(filename)[0]
        if file_id in ids:
            # Move the file to the corresponding output folder
            output_path1 = os.path.join(real_folder, file_id + '.jpg')
            shutil.copy(file_path, output_path1)
            print(f"Moved {filename} to {output_path1}")
        else:
            output_path2 = os.path.join(fake_folder, file_id + '.jpg')
            shutil.copy(file_path, output_path2)
            print(f"Moved {filename} to {output_path2}")

#Data Preprocessing
def proc_img(filepath):
    
    labels = list(map(lambda x: os.path.split(os.path.split(x)[0])[1], filepath))

    filepath = pd.Series(filepath, name='Filepath').astype(str)
    labels = pd.Series(labels, name='Label')

    df = pd.concat([filepath, labels], axis=1)

    df = df.sample(frac=1).reset_index(drop = True)
    
    return df
